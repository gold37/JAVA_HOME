 --게시판 글쓰기 테이블 생성 >>
                create table tbl_board  -- tbl_board는 테이블 명
                (boardno  number(3) -- boardno는 컬럼 명, 
                                                         /* 
                                                         number는 숫자 형태만 들어오는 데이터타입인데 
                                                         자릿수를 명기하지않으면 최대 38자리까지 표현이 가능하다.
                                                         그런데 자리수를 명기하면 표현이 아래와 같이 된다.
                                                         number(4) ==> -9999 ~ +9999까지 표현이 가능하다.
                                                         number(5) ==> -99999 ~ +99999까지 표현이 가능하다.
                                                         number(4,2) ==> -99.99 ~ +99.99까지 표현이 가능하다. (전체 자리수, 소수점 자리수)
                                                         */

, subject varchar2(2000) /*  
                                                 varchar2는 문자열 타입,  varchar2(4000)은 가능하지만  varchar2(4001)은 불가하다.
                                                숫자 4000은 byte크기를 말하는 것으로서 최대 4000 byte 까지 저장할 수 잇다.
                                                즉,  varchar2(1) ~  varchar2(4000)까지이다.
                                                그런데 한글은 한글 1글자가 2byte를 소모하므로  varchar2(4000)은 한글로 되어진 데이터는 2000글자까지만 저장가능하다.
                                                
                                                만약  varchar2(20)으로 되어진 칼럼에 데이터값으로  'kh정보교육원' 들어온다면 실제크기는 12byte만큼 소모된다.
                                                */
                                                
, content Nvarchar2(2000)  /* 
                                                      Nvarchar2(2000)에서 숫자 2000은 byte가 아니라 글자수를 의미한다.
                                                      만약에 varchar2(10)으로 되어진 컬럼에 데이터값으로  'kh정보교육원' 들어온다면 에러가 발생한다.
                                                      왜냐하면 최대 10byte까지만 허락하는데  'kh정보교육원'은 12byte이므로 용량이 초과되어 에러가 발생된다.
                                                      그런데 Nvarchar2(10)으로 되어진 컬럼에 데이터값으로  'kh정보교육원' 들어온다면 이것은 가능하다. 왜냐하면 이때 10은 byte가 아니라 글자 수이므로
                                                      최대 10글자까지는 허락하기 때문이다. 'kh정보교육원'은 7글자 이므로 .. !
                                                      Nvarchar2는 최대 2000글자까지만 허락한다.  
                                                    */

, content2 clob                        /*
                                                    clob은 Character Large object를 말하는 데이터 타입으로서 최대 4GB까지 문자열을 저장할 수 있다.
                                                  */
,userid varchar2(40)
,registerday date                   /* date는 날짜를 나타내는 타입이다. */
,readcount number(10)
);
